AIM :  To demonstrate the use of dynamic memory allocation and pointer manipulation  through a student record management system.

#include <iostream>
#include <string>
using namespace std;

// Structure for Student
struct Student {
    int rollNo;
    string name;
    float marks;
    Student* next;  // pointer to next student
};

// Class for Student Database
class StudentDatabase {
private:
    Student* head;

public:
    StudentDatabase() {
        head = nullptr;
    }

    // Insert new student
    void insertStudent(int roll, string name, float marks) {
        Student* newStudent = new Student();
        newStudent->rollNo = roll;
        newStudent->name = name;
        newStudent->marks = marks;
        newStudent->next = head;
        head = newStudent;
        cout << "Student added successfully!\n";
    }

    // Display all students
    void displayStudents() {
        if (head == nullptr) {
            cout << "No records found.\n";
            return;
        }
        Student* temp = head;
        cout << "\n--- Student Records ---\n";
        while (temp != nullptr) {
            cout << "Roll No: " << temp->rollNo
                 << ", Name: " << temp->name
                 << ", Marks: " << temp->marks << endl;
            temp = temp->next;
        }
    }

    // Search student by Roll No
    void searchStudent(int roll) {
        Student* temp = head;
        while (temp != nullptr) {
            if (temp->rollNo == roll) {
                cout << "Record Found -> Roll No: " << temp->rollNo
                     << ", Name: " << temp->name
                     << ", Marks: " << temp->marks << endl;
                return;
            }
            temp = temp->next;
        }
        cout << "Student with Roll No " << roll << " not found.\n";
    }

    // Delete student by Roll No
    void deleteStudent(int roll) {
        Student* temp = head;
        Student* prev = nullptr;

        while (temp != nullptr && temp->rollNo != roll) {
            prev = temp;
            temp = temp->next;
        }

        if (temp == nullptr) {
            cout << "Student not found.\n";
            return;
        }

        if (prev == nullptr) {
            head = temp->next; // deleting first student
        } else {
            prev->next = temp->next;
        }
        delete temp;
        cout << "Student deleted successfully.\n";
    }
};

// Driver Code
int main() {
    StudentDatabase db;
    int choice, roll;
    string name;
    float marks;

    do {
        cout << "\n--- Student Database Menu ---\n";
        cout << "1. Add Student\n2. Display All\n3. Search Student\n4. Delete Student\n5. Exit\n";
        cout << "Enter choice: ";
        cin >> choice;

        switch (choice) {
            case 1:
                cout << "Enter Roll No: "; cin >> roll;
                cout << "Enter Name: "; cin >> name;
                cout << "Enter Marks: "; cin >> marks;
                db.insertStudent(roll, name, marks);
                break;
            case 2:
                db.displayStudents();
                break;
            case 3:
                cout << "Enter Roll No to search: "; cin >> roll;
                db.searchStudent(roll);
                break;
            case 4:
                cout << "Enter Roll No to delete: "; cin >> roll;
                db.deleteStudent(roll);
                break;
            case 5:
                cout << "Exiting...\n";
                break;
            default:
                cout << "Invalid choice!\n";
        }
    } while (choice != 5);

    return 0;
}
